\textbf{–í—ã–≤–æ–¥ —É—Ä–∞–≤–Ω–µ–Ω–∏—è –¥–≤–∏–∂–µ–Ω–∏—è –º–∞—è—Ç–Ω–∏–∫–∞.} –†–∞—Å—Å–º–æ—Ç—Ä–∏–º —Å–∏–ª—ã, –¥–µ–π—Å—Ç–≤—É—é—â–∏–µ –Ω–∞ –º–∞—è—Ç–Ω–∏–∫. –û—Å–Ω–æ–≤–Ω–æ–π –≤—Ä–∞—â–∞—é—â–∏–π –º–æ–º–µ–Ω—Ç —Å–æ–∑–¥–∞—ë—Ç—Å—è –ø–æ–¥–≤–µ—à–µ–Ω–Ω—ã–º –Ω–∞ –Ω–∏—Ç–∏ –ø–µ—Ä–µ–≥—Ä—É–∑–∫–æ–º. –ù–µ–ø–æ—Å—Ä–µ–¥—Å—Ç–≤–µ–Ω–Ω–æ –Ω–∞ –º–∞—è—Ç–Ω–∏–∫ –¥–µ–π—Å—Ç–≤—É–µ—Ç –º–æ–º–µ–Ω—Ç —Å–∏–ª—ã –Ω–∞—Ç—è–∂–µ–Ω–∏—è –Ω–∏—Ç–∏: $M_\text{–Ω} = rT$, –≥–¥–µ $r$ --- —Ä–∞–¥–∏—É—Å —à–∫–∏–≤–∞ ($r_1$ –∏–ª–∏ $r_2$). –°–∏–ª—É $T$ –≤—ã—Ä–∞–∑–∏–º –∏–∑ —É—Ä–∞–≤–Ω–µ–Ω–∏—è –¥–≤–∏–∂–µ–Ω–∏—è –ø–ª–∞—Ç—Ñ–æ—Ä–º—ã: $m_\text{–Ω}y'' = m_\text{–Ω}g - T$, –≥–¥–µ $m_\text{–Ω} = m_\text{–ø} + m_\text{–≥}$ --- –º–∞—Å—Å–∞ –ø–ª–∞—Ç—Ñ–æ—Ä–º—ã —Å –ø–µ—Ä–µ–≥—Ä—É–∑–∫–æ–º. –£—Å–∫–æ—Ä–µ–Ω–∏–µ –ø–ª–∞—Ç—Ñ–æ—Ä–º—ã —Å–≤—è–∑–∞–Ω–æ —Å —É–≥–ª–æ–≤—ã–º —É—Å–∫–æ—Ä–µ–Ω–∏–µ–º –º–∞—è—Ç–Ω–∏–∫–∞ —É—Å–ª–æ–≤–∏–µ–º –Ω–µ—Ä–∞—Å—Ç—è–∂–∏–º–æ—Å—Ç–∏ –Ω–∏—Ç–∏ $y'' = \beta r$. –û—Ç—Å—é–¥–∞ –º–æ–º–µ–Ω—Ç —Å–∏–ª—ã –Ω–∞—Ç—è–∂–µ–Ω–∏—è –Ω–∏—Ç–∏

\begin{equation}
M_\text{–Ω} = m_\text{–Ω}r(g - \beta r)
\end{equation}

–í—Ä–∞—â–µ–Ω–∏—é –º–∞—è—Ç–Ω–∏–∫–∞ –ø—Ä–µ–ø—è—Ç—Å—Ç–≤—É–µ—Ç –º–æ–º–µ–Ω—Ç —Å–∏–ª—ã —Ç—Ä–µ–Ω–∏—è –≤ –æ—Å–∏ $M_\text{—Ç—Ä}$. –¢–∞–∫–∏–º –æ–±—Ä–∞–∑–æ–º, —Å —É—á–µ—Ç–æ–º (2) —É—Ä–∞–≤–Ω–µ–Ω–∏–µ (1) –º–æ–∂–µ—Ç –±—ã—Ç—å –∑–∞–ø–∏—Å–∞–Ω–æ –∫–∞–∫

\begin{equation}
(I + m_\text{–ù}r^2)\beta = m_\text{–Ω}gr - M_\text{—Ç—Ä}
\end{equation}

–ó–∞–º–µ—Ç–∏–º, —á—Ç–æ –≤ –Ω–∞—à–∏—Ö –æ–ø—ã—Ç–∞—Ö, –∫–∞–∫ –ø—Ä–∞–≤–∏–ª–æ, $m_\text{–Ω}r^2 \ll ùêº$, –∏ —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–µ–Ω–Ω–æ, $M_\text{–Ω} \approx m_\text{–Ω}gr$. –ï—Å–ª–∏ —Ç—Ä–µ–Ω–∏–µ –º–∞–ª–æ, $M_\text{—Ç—Ä} \ll m_\text{–Ω}gr$, —Ç–æ –º–∞—è—Ç–Ω–∏–∫ –±—É–¥–µ—Ç —Ä–∞—Å–∫—Ä—É—á–∏–≤–∞—Ç—å—Å—è —Å –ø–æ—Å—Ç–æ—è–Ω–Ω—ã–º —É–≥–ª–æ–≤—ã–º —É—Å–∫–æ—Ä–µ–Ω–∏–µ–º $\beta_0 \approx m_\text{–Ω}gr/I$.

$N = m_\text{–º}g + T \approx (m_\text{–º} + m_\text{–Ω})g \approx m_\text{–º}g$, –≥–¥–µ $m_\text{–º}$ ‚Äî –º–∞—Å—Å–∞ –º–∞—è—Ç–Ω–∏–∫–∞ (–∫–∞–∫ –ø—Ä–∞–≤–∏–ª–æ,
